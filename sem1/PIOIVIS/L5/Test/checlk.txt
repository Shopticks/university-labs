for %%f in ("%folder%*") do (
    set temp=%folder%temp.txt
    echo. > %temp%
    echo -----------------------------
    echo Check file: %%f

    for /f "delims=" %%A in ('type "%%f"') do (
        set "string=%%A"
        set "modified=!string:123=321!"
        echo !modified!>>"%temp%"
    )
    move %temp% %%f
    rm -rf %temp%
)
